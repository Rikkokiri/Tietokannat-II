CREATE TABLE PELAAJA(
	ID INT PRIMARY KEY NOT NULL,
	NIMI TEXT NOT NULL,
	PUHNUM INT NOT NULL,
	KOTIPAIKKA CHAR(30) NOT NULL
);

CREATE TABLE RATA(
	ID INT PRIMARY KEY NOT NULL,
	LUOKITUS CHAR(4) NOT NULL,
	VAYLIEN_LKM INT NOT NULL,
	OSOITE CHAR(40) NOT NULL,
	RATAMESTARI TEXT NOT NULL
);

CREATE TABLE PELI(
	ID INT PRIMARY KEY NOT NULL,
	RADAN_ID INT NOT NULL,
	PAIVAMAARA CHAR(10) NOT NULL,
	FOREIGN KEY(RADAN_ID) REFERENCES RATA(ID)
);

CREATE TABLE PELAAMASSA(
	PELIID INT NOT NULL,
	PELAAJAID INT NOT NULL,
	FOREIGN KEY(PELIID) REFERENCES PELI(ID),
	FOREIGN KEY(PELAAJAID) REFERENCES PELAAJA(ID)
);

CREATE TABLE SUORITUS(
	PELAAJAID INT NOT NULL,
	PELIID INT NOT NULL,
	RADANID INT NOT NULL,
	VAYLANNUMERO INT NOT NULL,
	HEITTOJEN_LKM INT NOT NULL,
	FOREIGN KEY(PELAAJAID) REFERENCES PELAAJA(ID),
	FOREIGN KEY(PELIID) REFERENCES PELI(ID),
	FOREIGN KEY(RADANID) REFERENCES RATA(ID)
);

CREATE TABLE VAYLA(
	RADANID INT NOT NULL,
	PAR INT NOT NULL,
	NUMERO INT NOT NULL,
	PITUUS INT NOT NULL,
	FOREIGN KEY(RADANID) REFERENCES RATA(ID)
);
